\section{Experience}

\begin{luacode}
  for k, exp in ipairs(resume['experience']) do

    local items = ''
    for j,bullet in ipairs(exp['bulletPoints']) do
      items = items .. string.format('\\item \%s', bullet)
    end

    local vsp = ''
    if k == 1 then
      vsp = '\\vspace{\\topsep}'
    end

    tex.sprint(
      string.format(
	'\\runsubsection{\%s}' ..
	'\\descript{| \%s}' ..
	'\\location{\%s - \%s | \%s}' ..
	vsp ..
	'\\begin{tightemize}' ..
	'\\sectionsep' ..
	'\%s' ..
	'\\end{tightemize}' ..
	'\\sectionsep' ..
	'\\vspace{3mm}',
	exp['company'],
	exp['jobTitle'],
	exp['start'],
	exp['end'],
	exp['location'],
	items
      )
    )
  end
\end{luacode}
